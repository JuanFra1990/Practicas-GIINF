MVI B,00
MVI C,00
MVI D,00
MVI E,00
MVI H,00
MVI L,00
OUT 02
;Comienza el programa
IN 00
MOV B,A
LDAX B
MOV B,A
IN 01
MOV E,A
LDAX D
MOV D,A
STA 0001
CMP B
MOV A,D
JC ERROR

CONTADOR:MOV A,E
ADI 01
MOV E,A
LDA 0001
SUI 01
CPI 00
JZ ERROR
STA 0001
JNZ COMPROBAR
COMPROBAR: MVI B,00
MVI D,00
LDAX B
MOV H,A
LDAX D
MOV L,A
MOV A,L
CMP H
JZ COMPROBAR2
JNZ CONTADOR
COMPROBAR2: MVI A,00
HLT
ERROR: MVI A,FF
