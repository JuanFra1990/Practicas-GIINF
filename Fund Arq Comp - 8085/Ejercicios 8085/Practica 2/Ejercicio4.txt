;De una lista de 5 comprueba si el numero depositado es par de paridad, o impar, si es impar le suma 80 (En binario el 128, bit mas a la izquierda)
LDA 0100
MOV L,A
MVI H,01
CALL DATO1
CALL CONTADOR
CALL DATO2
CALL DATO3
CALL DATO4
CALL DATO5

CONTADOR:MOV A,L
SUB H
JZ NOCONTINUA
MOV L,A
RET

IMPAR: MVI D,80
ADD D
STA 0101
RET
PAR: STA 0101
RET

DATO1:LDA 0101
CPO IMPAR
CPE PAR
RET

DATO2:LDA 0102
CPO IMPAR
CPE PAR
CALL CONTADOR
RET

DATO3:LDA 0103
CPO IMPAR
CPE PAR
CALL CONTADOR
RET

DATO4:LDA 0104
CPO IMPAR
CPE PAR
CALL CONTADOR
RET

DATO5:LDA 0105
CPO IMPAR
CPE PAR
CALL CONTADOR
RET

NOCONTINUA:HLT
