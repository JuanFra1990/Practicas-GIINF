MVI C,00
MVI D,00
MVI E,F1
IN 00
MOV B,A
LDA 00F0
SUI 01
MOV H,A
JC FALLO
JNC CARGARDATO
CONTINUA:MOV A,E
ADI 01
MOV E,A
MOV A,H
SUI 01
JNC CARGARDATO
JC RESULTADO
CARGARDATO:MOV H,A
 LDAX D        
CMP B
JZ MISMO
CALL CONTINUA
MISMO: MOV L,E
MVI A,00
ADI 01
MOV C,A
CALL CONTINUA
FALLO: MVI A,FF
OUT 02
HLT
RESULTADO: MOV A,C
CMP D
JZ FALLO
MOV A,L
OUT 02
HLT
