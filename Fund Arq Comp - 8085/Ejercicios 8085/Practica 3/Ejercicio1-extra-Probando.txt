;Limpiar los registros
MVI A,00
MOV B,A
MOV C,A
MOV D,A
MOV E,A
MOV H,A
MOV L,A
;Hacemos la division exacta
MVI E,06
LDA 0100;Dividendo
ADD E
MOV B,A
MOV C,A
LDA 0101;Divisor
ADD E
MOV L,A
MVI H,01
CONTINUA: MOV A,D
ADD H
MOV D,A
MOV A,B
SUB L
CMP L
MOV B,A
CC NOCONTINUA
CNC CONTINUA
NOCONTINUA:MVI L,06
CMP L
JC NOAJUSTE
JNC AJUSTE
NOAJUSTE:STA 0103
JMP RESULTADO
AJUSTE: SUB E
STA 0103
JMP RESULTADO
RESULTADO:MOV A,D
ADD E
STA 0102
HLT
