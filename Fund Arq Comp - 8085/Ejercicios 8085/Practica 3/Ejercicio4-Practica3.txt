MVI A,00
OUT 01
OUT 02
OUT 03
OUT 04
MVI B,00
MVI C,00
MVI D,00
MVI E,F1
IN 00
SUI 01
MOV H,A
JC FALLO
JNC CARGARDATO
CONTINUA:MOV A,E
ADI 01
MOV E,A
MOV A,H
SUI 01
JNC CARGARDATO
JC RESULTADO
CARGARDATO:MOV H,A
LDAX D        
ADD B 
MOV B,A
CALL CONTINUA

FALLO: MVI A,FF
OUT 00
OUT 02
OUT 04
MVI A,80
OUT 01
OUT 03
HLT
RESULTADO:IN 00
MOV C,A
CALL DIVISION
MOV A,H
OUT 02
HLT
EXACTA: RET
DIVISION: MOV A,H
ADI 01
MOV H,A
MOV A,B
SUB C
MOV B,A
JZ EXACTA
JNC DIVISION
RET
